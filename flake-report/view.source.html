253 def ask_date(self, question, time: bool = False):
254  
255 format = "%d/%m/%Y"
256 date = input(question)
257 if time:
258 format = format + " %H:%M"
259 try:
260 date = datetime.strptime(date, format)
261 if not time:
262 date = date.date()
263 except ValueError:
264 error ="La date doit etre au format jj/mm/aaaa "
265 if time:
266 error = error + "heure:minute"
